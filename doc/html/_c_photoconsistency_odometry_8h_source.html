00045 void eigenPose(float x,
00046            float y,
00047            float z,
00048            float yaw,
00049            float pitch,
00050            float roll,
00051            Eigen::Matrix4f & pose)
00052 {
00053     pose(0,0) = cos(yaw) * cos(pitch);
00054     pose(0,1) = cos(yaw) * sin(pitch) * sin(roll) - sin(yaw) * cos(roll);
00055     pose(0,2) = cos(yaw) * sin(pitch) * cos(roll) + sin(yaw) * sin(roll);
00056     pose(0,3) = x;
00057 
00058     pose(1,0) = sin(yaw) * cos(pitch);
00059     pose(1,1) = sin(yaw) * sin(pitch) * sin(roll) + cos(yaw) * cos(roll);
00060     pose(1,2) = sin(yaw) * sin(pitch) * cos(roll) - cos(yaw) * sin(roll);
00061     pose(1,3) = y;
00062 
00063     pose(2,0) = -sin(pitch);
00064     pose(2,1) = cos(pitch) * sin(roll);
00065     pose(2,2) = cos(pitch) * cos(roll);
00066     pose(2,3) = z;
00067 
00068     pose(3,0) = 0;
00069     pose(3,1) = 0;
00070     pose(3,2) = 0;
00071     pose(3,3) = 1;

00075 template <class T>
00076 void warpImage(cv::Mat & imgGray,
00077                cv::Mat & imgDepth,
00078                cv::Mat & imgGrayWarped,
00079                Eigen::Matrix4f & Rt,
00080                Eigen::Matrix3f & cameraMatrix,int level=0)
00081 {
00082     float fx = cameraMatrix(0,0)/pow(2,level);
00083     float fy = cameraMatrix(1,1)/pow(2,level);
00084     float inv_fx = 1.f/fx;
00085     float inv_fy = 1.f/fy;
00086     float ox = cameraMatrix(0,2)/pow(2,level);
00087     float oy = cameraMatrix(1,2)/pow(2,level);
00088 
00089     Eigen::Vector4f point3D;
00090     Eigen::Vector4f transformedPoint3D;
00091     int transformed_r,transformed_c; // 2D coordinates of the transformed pixel(r,c) of frame 1
00092 
00093     imgGrayWarped = cv::Mat::zeros(imgGray.rows,imgGray.cols,imgGray.type());
00094 
00095     #if ENABLE_OPENMP_MULTITHREADING_WARP_IMAGE
00096     #pragma omp parallel for private(point3D,transformedPoint3D,transformed_r,transformed_c)
00097     #endif
00098     for(int r=0;r<imgGray.rows;r++)
00099     {
00100         for(int c=0;c<imgGray.cols;c++)
00101         {
00102             if(imgDepth.at<float>(r,c)>0) //If has valid depth value
00103             {
00104                 //Compute the local 3D coordinates of pixel(r,c) of frame 1
00105                 point3D(2) = imgDepth.at<float>(r,c); //z
00106                 point3D(0) = (c-ox) * point3D(2) * inv_fx;         //x
00107                 point3D(1) = (r-oy) * point3D(2) * inv_fy;         //y
00108                 point3D(3) = 1.0;                          //homogeneous coordinate
00109 
00110                 //Transform the 3D point using the transformation matrix Rt
00111                 transformedPoint3D = Rt * point3D;
00112 
00113                 //Project the 3D point to the 2D plane
00114                 transformed_c = ((transformedPoint3D(0) * fx) / transformedPoint3D(2)) + ox; //transformed x (2D)
00115                 transformed_r = ((transformedPoint3D(1) * fy) / transformedPoint3D(2)) + oy; //transformed y (2D)
00116 
00117                 //Asign the intensity value to the warped image and compute the difference between the transformed
00118                 //pixel of frame 1 and the corresponding pixel of frame 2. Compute the error function
00119                 if(transformed_r>=0 && transformed_r < imgGray.rows &
00120                    transformed_c>=0 && transformed_c < imgGray.cols)
00121                 {
00122                     imgGrayWarped.at<T>(transformed_r,transformed_c)=imgGray.at<T>(r,c);
00123                 }
00124             }
00125         }
00126     }
00127 }
