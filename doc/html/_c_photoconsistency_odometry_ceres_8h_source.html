00079            float y,
00080            float z,
00081            float yaw,
00082            float pitch,
00083            float roll,
00084            Eigen::Matrix4f & pose)
00085 {
00086     pose(0,0) = cos(yaw) * cos(pitch);
00087     pose(0,1) = cos(yaw) * sin(pitch) * sin(roll) - sin(yaw) * cos(roll);
00088     pose(0,2) = cos(yaw) * sin(pitch) * cos(roll) + sin(yaw) * sin(roll);
00089     pose(0,3) = x;
00090 
00091     pose(1,0) = sin(yaw) * cos(pitch);
00092     pose(1,1) = sin(yaw) * sin(pitch) * sin(roll) + cos(yaw) * cos(roll);
00093     pose(1,2) = sin(yaw) * sin(pitch) * cos(roll) - cos(yaw) * sin(roll);
00094     pose(1,3) = y;
00095 
00096     pose(2,0) = -sin(pitch);
00097     pose(2,1) = cos(pitch) * sin(roll);
00098     pose(2,2) = cos(pitch) * cos(roll);
00099     pose(2,3) = z;
00100 
00101     pose(3,0) = 0;
00102     pose(3,1) = 0;

00108 template <class T>
00109 void warpImage(cv::Mat & imgGray,
00110                cv::Mat & imgDepth,
00111                cv::Mat & imgGrayWarped,
00112                Eigen::Matrix4f & Rt,
00113                Eigen::Matrix3f & cameraMatrix,int level=0)
00114 {
00115     float fx = cameraMatrix(0,0)/pow(2,level);
00116     float fy = cameraMatrix(1,1)/pow(2,level);
00117     float inv_fx = 1.f/fx;
00118     float inv_fy = 1.f/fy;
00119     float ox = cameraMatrix(0,2)/pow(2,level);
00120     float oy = cameraMatrix(1,2)/pow(2,level);
00121 
00122     Eigen::Vector4f point3D;
00123     Eigen::Vector4f transformedPoint3D;
00124     int transformed_r,transformed_c; // 2D coordinates of the transformed pixel(r,c) of frame 1
00125 
00126     imgGrayWarped = cv::Mat::zeros(imgGray.rows,imgGray.cols,imgGray.type());
00127 
00128     #if ENABLE_OPENMP_MULTITHREADING
00129     #pragma omp parallel for private(point3D,transformedPoint3D,transformed_r,transformed_c)
00130     #endif
00131     for(int r=0;r<imgGray.rows;r++)
00132     {
00133         for(int c=0;c<imgGray.cols;c++)
00134         {
00135             if(imgDepth.at<float>(r,c)>0) //If has valid depth value
00136             {
00137                 //Compute the local 3D coordinates of pixel(r,c) of frame 1
00138                 point3D(2) = imgDepth.at<float>(r,c); //z
00139                 point3D(0) = (c-ox) * point3D(2) * inv_fx;         //x
00140                 point3D(1) = (r-oy) * point3D(2) * inv_fy;         //y
00141                 point3D(3) = 1.0;                          //homogeneous coordinate
00142 
00143                 //Transform the 3D point using the transformation matrix Rt
00144                 transformedPoint3D = Rt * point3D;
00145 
00146                 //Project the 3D point to the 2D plane
00147                 transformed_c = ((transformedPoint3D(0) * fx) / transformedPoint3D(2)) + ox; //transformed x (2D)
00148                 transformed_r = ((transformedPoint3D(1) * fy) / transformedPoint3D(2)) + oy; //transformed y (2D)
00149 
00150                 //Asign the intensity value to the warped image and compute the difference between the transformed
00151                 //pixel of frame 1 and the corresponding pixel of frame 2. Compute the error function
00152                 if(transformed_r>=0 && transformed_r < imgGray.rows &
00153                    transformed_c>=0 && transformed_c < imgGray.cols)
00154                 {
00155                     imgGrayWarped.at<T>(transformed_r,transformed_c)=imgGray.at<T>(r,c);
00156                 }
00157             }
00158         }
00159     }

00183 template<typename T>
00184 void SampleLinear(const cv::Mat & intensityImage,
00185                   const cv::Mat & intensityGradientX,
00186                   const cv::Mat & intensityGradientY,
00187                   double y, double x, T* sample) {
00188 
00189   int x1, y1, x2, y2;
00190   double dx, dy;
00191 
00192   // Take the upper left corner as integer pixel positions.
00193   x -= 0.5;
00194   y -= 0.5;
00195 
00196   LinearInitAxis(y, intensityImage.rows, &y1, &y2, &dy);
00197   LinearInitAxis(x, intensityImage.cols,  &x1, &x2, &dx);
00198 
00199   //Sample intensity
00200   const T im11 = T(intensityImage.at<float>(y1, x1));
00201   const T im12 = T(intensityImage.at<float>(y1, x2));
00202   const T im21 = T(intensityImage.at<float>(y2, x1));
00203   const T im22 = T(intensityImage.at<float>(y2, x2));
00204 
00205   sample[0] =(     dy  * ( dx * im11 + (1.0 - dx) * im12 ) +
00206            (1 - dy) * ( dx * im21 + (1.0 - dx) * im22 ));
00207 
00208   //Sample gradient x
00209   const T gradx11 = T(intensityGradientX.at<float>(y1, x1));
00210   const T gradx12 = T(intensityGradientX.at<float>(y1, x2));
00211   const T gradx21 = T(intensityGradientX.at<float>(y2, x1));
00212   const T gradx22 = T(intensityGradientX.at<float>(y2, x2));
00213 
00214   sample[1] =(     dy  * ( dx * gradx11 + (1.0 - dx) * gradx12 ) +
00215            (1 - dy) * ( dx * gradx21 + (1.0 - dx) * gradx22 ));
00216 
00217   //Sample gradient y
00218   const T grady11 = T(intensityGradientY.at<float>(y1, x1));
00219   const T grady12 = T(intensityGradientY.at<float>(y1, x2));
00220   const T grady21 = T(intensityGradientY.at<float>(y2, x1));
00221   const T grady22 = T(intensityGradientY.at<float>(y2, x2));
00222 
00223   sample[2] =(     dy  * ( dx * grady11 + (1.0 - dx) * grady12 ) +
00224            (1 - dy) * ( dx * grady21 + (1.0 - dx) * grady22 ));
00225 }

00227 // Sample the image at position (x, y) but use the gradient to
00228 // propagate derivatives from x and y. This is needed to integrate the numeric
00229 // image gradients with Ceres's autodiff framework.
00230 template<typename T>
00231 T SampleWithDerivative(const cv::Mat & intensityImage,
00232                               const cv::Mat & intensityGradientX,
00233                               const cv::Mat & intensityGradientY,
00234                               const T & x,
00235                               const T & y)
00236 {
00237   double scalar_x = ceres::JetOps<T>::GetScalar(x);
00238   double scalar_y = ceres::JetOps<T>::GetScalar(y);
00239 
00240   double sample[3];
00241   // Sample intensity image and gradients
00242   SampleLinear(intensityImage,intensityGradientX,intensityGradientY, scalar_y, scalar_x, sample);
00243   T xy[2] = { x, y };
00244   return ceres::Chain<double, 2, T>::Rule(sample[0], sample + 1, xy);
00245 }

00406     void buildPyramid(cv::Mat & img,std::vector<cv::Mat>& pyramid,int levels,bool applyBlur)
00407     {
00408         //Create space for all the images
00409         pyramid.resize(levels);
00410 
00411         double factor = 1;
00412         for(int level=0;level<levels;level++)
00413         {
00414             //Create an auxiliar image of factor times the size of the original image
00415             cv::Mat imgAux;
00416             if(level!=0)
00417             {
00418                 cv::resize(img,imgAux,cv::Size(0,0),factor,factor);
00419             }
00420             else
00421             {
00422                 imgAux = img;
00423             }
00424 
00425             //Blur the resized image with different filter size depending on the current pyramid level
00426             if(applyBlur)
00427             {
00428                 #if ENABLE_GAUSSIAN_BLUR
00429                 if(blurFilterSize[level]>0)
00430                 {
00431                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00432                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00433                 }
00434                 #elif ENABLE_BOX_FILTER_BLUR
00435                 if(blurFilterSize[level]>0)
00436                 {
00437                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00438                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00439                 }
00440                 #endif
00441             }
00442 
00443             //Assign the resized image to the current level of the pyramid
00444             pyramid[level]=imgAux;
00445 
00446             factor = factor/2;
00447         }
00448     }
