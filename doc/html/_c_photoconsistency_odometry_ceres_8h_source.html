00095       template <typename T> bool operator()(const T* const stateVector,
00096                                             T* residuals) const {
00097 
00098         //Set camera parameters depending on the optimization level
00099         T fx = T(cameraMatrix[0][0])/pow(2,T(optimizationLevel));
00100         T fy = T(cameraMatrix[1][1])/pow(2,T(optimizationLevel));
00101         T inv_fx = T(1)/fx;
00102         T inv_fy = T(1)/fy;
00103         T ox = T(cameraMatrix[0][2])/pow(2,T(optimizationLevel));
00104         T oy = T(cameraMatrix[1][2])/pow(2,T(optimizationLevel));
00105 
00106         //Compute the rigid transformation matrix from the parameters
00107         T x = stateVector[0];
00108         T y = stateVector[1];
00109         T z = stateVector[2];
00110         T yaw = stateVector[3];
00111         T pitch = stateVector[4];
00112         T roll = stateVector[5];
00113         T Rt[4][4];
00114         T sin_yaw = sin(yaw);
00115         T cos_yaw = cos(yaw);
00116         T sin_pitch = sin(pitch);
00117         T cos_pitch = cos(pitch);
00118         T sin_roll = sin(roll);
00119         T cos_roll = cos(roll);
00120         Rt[0][0] = cos_yaw * cos_pitch;
00121         Rt[0][1] = cos_yaw * sin_pitch * sin_roll - sin_yaw * cos_roll;
00122         Rt[0][2] = cos_yaw * sin_pitch * cos_roll + sin_yaw * sin_roll;
00123         Rt[0][3] = x;
00124         Rt[1][0] = sin_yaw * cos_pitch;
00125         Rt[1][1] = sin_yaw * sin_pitch * sin_roll + cos_yaw * cos_roll;
00126         Rt[1][2] = sin_yaw * sin_pitch * cos_roll - cos_yaw * sin_roll;
00127         Rt[1][3] = y;
00128         Rt[2][0] = -sin_pitch;
00129         Rt[2][1] = cos_pitch * sin_roll;
00130         Rt[2][2] = cos_pitch * cos_roll;
00131         Rt[2][3] = z;
00132         Rt[3][0] = T(0);
00133         Rt[3][1] = T(0);
00134         Rt[3][2] = T(0);
00135         Rt[3][3] = T(1);
00136 
00137         //Initialize the error function (residuals) with an initial value
00138         #if ENABLE_OPENMP_MULTITHREADING_CERES
00139         #pragma omp parallel for
00140         #endif
00141         for(int r=0;r<gray0Pyr[optimizationLevel].rows;r++)
00142         {
00143             for(int c=0;c<gray0Pyr[optimizationLevel].cols;c++)
00144             {
00145                 residuals[gray0Pyr[optimizationLevel].cols*r+c]=T(0);
00146             }
00147         }
00148 
00149         T residualScalingFactor = T(1);
00150 
00151         #if ENABLE_OPENMP_MULTITHREADING_CERES
00152         #pragma omp parallel for
00153         #endif
00154         for(int r=0;r<gray0Pyr[optimizationLevel].rows;r++)
00155         {
00156 
00157             T point3D[4];
00158             T transformedPoint3D[4];
00159             T transformed_r,transformed_c; // 2D coordinates of the transformed pixel(r,c) of frame 1
00160             T pixel1; //Intensity value of the pixel(r,c) of the warped frame 1
00161             T pixel2; //Intensity value of the pixel(r,c) of frame 2
00162 
00163             for(int c=0;c<gray0Pyr[optimizationLevel].cols;c++)
00164             {
00165                 if(depth0Pyr[optimizationLevel].at<float>(r,c)>0) //If has valid depth value
00166                 {
00167                     //Compute the local 3D coordinates of pixel(r,c) of frame 1
00168                     point3D[2] = T(depth0Pyr[optimizationLevel].at<float>(r,c)); //z
00169                     point3D[0] = (T(c)-ox) * point3D[2] * inv_fx;          //x
00170                     point3D[1] = (T(r)-oy) * point3D[2] * inv_fy;          //y
00171                     point3D[3] = T(1.0);                           //homogeneous coordinate
00172 
00173                     //Transform the 3D point using the transformation matrix Rt
00174                     transformedPoint3D[0] = Rt[0][0]*point3D[0]+Rt[0][1]*point3D[1]+Rt[0][2]*point3D[2]+Rt[0][3]*point3D[3];
00175                     transformedPoint3D[1] = Rt[1][0]*point3D[0]+Rt[1][1]*point3D[1]+Rt[1][2]*point3D[2]+Rt[1][3]*point3D[3];
00176                     transformedPoint3D[2] = Rt[2][0]*point3D[0]+Rt[2][1]*point3D[1]+Rt[2][2]*point3D[2]+Rt[2][3]*point3D[3];
00177                     transformedPoint3D[3] = Rt[3][0]*point3D[0]+Rt[3][1]*point3D[1]+Rt[3][2]*point3D[2]+Rt[3][3]*point3D[3];
00178 
00179                     //Project the 3D point to the 2D plane
00180                     transformed_c = ((transformedPoint3D[0] * fx) / transformedPoint3D[2]) + ox; //transformed x (2D)
00181                     transformed_r = ((transformedPoint3D[1] * fy) / transformedPoint3D[2]) + oy; //transformed y (2D)
00182 
00183                     //Asign the intensity value to the warped image and compute the difference between the transformed
00184                     //pixel of frame 1 and the corresponding pixel of frame 2. Compute the error function
00185                     if(transformed_r>=T(0) && transformed_r < T(gray0Pyr[optimizationLevel].rows) &
00186                        transformed_c>=T(0) && transformed_c < T(gray0Pyr[optimizationLevel].cols))
00187                     {
00188                         //Compute the proyected coordinates of the transformed 3D point
00189                         int transformed_r_scalar = static_cast<int>(ceres::JetOps<T>::GetScalar(transformed_r));
00190                         int transformed_c_scalar = static_cast<int>(ceres::JetOps<T>::GetScalar(transformed_c));
00191 
00192                         //Compute the pixel residual
00193                         pixel1 = T(gray0Pyr[optimizationLevel].at<float>(r,c));
00194                         pixel2 = SampleWithDerivative(gray1Pyr[optimizationLevel],
00195                                                       gray1GradXPyr[optimizationLevel],
00196                                                       gray1GradYPyr[optimizationLevel],transformed_c,transformed_r);
00197                         residuals[gray0Pyr[optimizationLevel].cols*transformed_r_scalar+transformed_c_scalar] = residualScalingFactor * (pixel1 - pixel2);
00198 
00199                     }
00200                 }
00201             }
00202         }
00203 
00204         return true;
00205       }

00245     void buildPyramid(cv::Mat & img,std::vector<cv::Mat>& pyramid,int levels,bool applyBlur)
00246     {
00247         //Create space for all the images
00248         pyramid.resize(levels);
00249 
00250         double factor = 1;
00251         for(int level=0;level<levels;level++)
00252         {
00253             //Create an auxiliar image of factor times the size of the original image
00254             cv::Mat imgAux;
00255             if(level!=0)
00256             {
00257                 cv::resize(img,imgAux,cv::Size(0,0),factor,factor);
00258             }
00259             else
00260             {
00261                 imgAux = img;
00262             }
00263 
00264             //Blur the resized image with different filter size depending on the current pyramid level
00265             if(applyBlur)
00266             {
00267                 #if ENABLE_GAUSSIAN_BLUR
00268                 if(blurFilterSize[level]>0)
00269                 {
00270                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00271                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00272                 }
00273                 #elif ENABLE_BOX_FILTER_BLUR
00274                 if(blurFilterSize[level]>0)
00275                 {
00276                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00277                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00278                 }
00279                 #endif
00280             }
00281 
00282             //Assign the resized image to the current level of the pyramid
00283             pyramid[level]=imgAux;
00284 
00285             factor = factor/2;
00286         }
00287     }

00369     void optimize()
00370     {
00371         for(optimizationLevel = numOptimizationLevels-1;optimizationLevel>=0;optimizationLevel--)
00372         {
00373             // Build the problem.
00374             Problem problem;
00375 
00376             // Set up the only cost function (also known as residual). This uses
00377             // auto-differentiation to obtain the derivative (jacobian).
00378             problem.AddResidualBlock(
00379             new AutoDiffCostFunction<ResidualRGBDPhotoconsistency,ceres::DYNAMIC,6>(
00380               new ResidualRGBDPhotoconsistency,
00381               gray0Pyr[optimizationLevel].cols*gray0Pyr[optimizationLevel].rows /*dynamic size*/),
00382             NULL,
00383             x);
00384 
00385             // Run the solver!
00386             Solver::Options options;
00387             options.max_num_iterations = max_num_iterations[optimizationLevel];
00388             options.linear_solver_type = ceres::SPARSE_NORMAL_CHOLESKY;//ceres::DENSE_QR;

00396             options.min_relative_decrease = min_relative_decrease[optimizationLevel];
00397             options.num_linear_solver_threads = num_linear_solver_threads;

00400             VisualizationCallback callback;

00408                 options.update_state_every_iteration = false;
