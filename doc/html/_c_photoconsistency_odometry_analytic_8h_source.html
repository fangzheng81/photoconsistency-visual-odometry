00093     void buildPyramid(cv::Mat & img,std::vector<cv::Mat>& pyramid,int levels,bool applyBlur)
00094     {
00095         //Create space for all the images
00096         pyramid.resize(levels);
00097 
00098         double factor = 1;
00099         for(int level=0;level<levels;level++)
00100         {
00101             //Create an auxiliar image of factor times the size of the original image
00102             cv::Mat imgAux;
00103             if(level!=0)
00104             {
00105                 cv::resize(img,imgAux,cv::Size(0,0),factor,factor);
00106             }
00107             else
00108             {
00109                 imgAux = img;
00110             }
00111 
00112             //Blur the resized image with different filter size depending on the current pyramid level
00113             if(applyBlur)
00114             {
00115                 #if ENABLE_GAUSSIAN_BLUR
00116                 if(blurFilterSize[level]>0)
00117                 {
00118                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00119                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00120                 }
00121                 #elif ENABLE_BOX_FILTER_BLUR
00122                 if(blurFilterSize[level]>0)
00123                 {
00124                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00125                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00126                 }
00127                 #endif
00128             }
00129 
00130             //Assign the resized image to the current level of the pyramid
00131             pyramid[level]=imgAux;
00132 
00133             factor = factor/2;
00134         }
00135     }

00160     void computeResidualsAndJacobians(cv::Mat & source_grayImg,
00161                                       cv::Mat & source_depthImg,
00162                                       cv::Mat & target_grayImg,
00163                                       cv::Mat & target_gradXImg,
00164                                       cv::Mat & target_gradYImg,
00165                                       Eigen::Matrix<double,Eigen::Dynamic,1> & residuals,
00166                                       Eigen::Matrix<double,Eigen::Dynamic,6> & jacobians,
00167                                       cv::Mat & warped_source_grayImage)
00168     {
00169         int nRows = source_grayImg.rows;
00170         int nCols = source_grayImg.cols;
00171 
00172         double scaleFactor = 1.0/pow(2,optimizationLevel);
00173         double fx = cameraMatrix(0,0)*scaleFactor;
00174         double fy = cameraMatrix(1,1)*scaleFactor;
00175         double ox = cameraMatrix(0,2)*scaleFactor;
00176         double oy = cameraMatrix(1,2)*scaleFactor;
00177         float inv_fx = 1.f/fx;
00178         float inv_fy = 1.f/fy;
00179 
00180         double x = stateVector[0];
00181         double y = stateVector[1];
00182         double z = stateVector[2];
00183         double yaw = stateVector[3];
00184         double pitch = stateVector[4];
00185         double roll = stateVector[5];
00186 
00187         //Compute the rigid transformation matrix from the parameters
00188         Eigen::Matrix4f Rt = Eigen::Matrix4f::Identity();
00189         double sin_yaw = sin(yaw);
00190         double cos_yaw = cos(yaw);
00191         double sin_pitch = sin(pitch);
00192         double cos_pitch = cos(pitch);
00193         double sin_roll = sin(roll);
00194         double cos_roll = cos(roll);
00195         Rt(0,0) = cos_yaw * cos_pitch;
00196         Rt(0,1) = cos_yaw * sin_pitch * sin_roll - sin_yaw * cos_roll;
00197         Rt(0,2) = cos_yaw * sin_pitch * cos_roll + sin_yaw * sin_roll;
00198         Rt(0,3) = x;
00199         Rt(1,0) = sin_yaw * cos_pitch;
00200         Rt(1,1) = sin_yaw * sin_pitch * sin_roll + cos_yaw * cos_roll;
00201         Rt(1,2) = sin_yaw * sin_pitch * cos_roll - cos_yaw * sin_roll;
00202         Rt(1,3) = y;
00203         Rt(2,0) = -sin_pitch;
00204         Rt(2,1) = cos_pitch * sin_roll;
00205         Rt(2,2) = cos_pitch * cos_roll;
00206         Rt(2,3) = z;
00207         Rt(3,0) = 0;
00208         Rt(3,1) = 0;
00209         Rt(3,2) = 0;
00210         Rt(3,3) = 1;
00211 
00212         double temp1 = cos(pitch)*sin(roll);
00213         double temp2 = cos(pitch)*cos(roll);
00214         double temp3 = sin(pitch);
00215         double temp4 = (sin(roll)*sin(yaw)+sin(pitch)*cos(roll)*cos(yaw));
00216         double temp5 = (sin(pitch)*sin(roll)*cos(yaw)-cos(roll)*sin(yaw));
00217         double temp6 = (sin(pitch)*sin(roll)*sin(yaw)+cos(roll)*cos(yaw));
00218         double temp7 = (-sin(pitch)*sin(roll)*sin(yaw)-cos(roll)*cos(yaw));
00219         double temp8 = (sin(roll)*cos(yaw)-sin(pitch)*cos(roll)*sin(yaw));
00220         double temp9 = (sin(pitch)*cos(roll)*sin(yaw)-sin(roll)*cos(yaw));
00221         double temp10 = cos(pitch)*sin(roll)*cos(yaw);
00222         double temp11 = cos(pitch)*cos(yaw)+x;
00223         double temp12 = cos(pitch)*cos(roll)*cos(yaw);
00224         double temp13 = sin(pitch)*cos(yaw);
00225         double temp14 = cos(pitch)*sin(yaw);
00226         double temp15 = cos(pitch)*cos(yaw);
00227         double temp16 = sin(pitch)*sin(roll);
00228         double temp17 = sin(pitch)*cos(roll);
00229         double temp18 = cos(pitch)*sin(roll)*sin(yaw);
00230         double temp19 = cos(pitch)*cos(roll)*sin(yaw);
00231         double temp20 = sin(pitch)*sin(yaw);
00232         double temp21 = (cos(roll)*sin(yaw)-sin(pitch)*sin(roll)*cos(yaw));
00233         double temp22 = cos(pitch)*cos(roll);
00234         double temp23 = cos(pitch)*sin(roll);
00235         double temp24 = cos(pitch);
00236 
00237         #if ENABLE_OPENMP_MULTITHREADING_ANALYTIC
00238         #pragma omp parallel for
00239         #endif
00240         for (int r=0;r<nRows;r++)
00241         {
00242             for (int c=0;c<nCols;c++)
00243             {
00244                 int i = nCols*r+c; //vector index
00245 
00246                 //Compute the 3D coordinates of the pij of the source frame
00247                 Eigen::Vector4f point3D;
00248                 point3D(2)=source_depthImg.at<float>(r,c);
00249                 if(minDepth < point3D(2) && point3D(2) < maxDepth)//Compute the jacobian only for the valid points
00250                 {
00251                     point3D(0)=(c - ox) * point3D(2) * inv_fx;
00252                     point3D(1)=(r - oy) * point3D(2) * inv_fy;
00253                     point3D(3)=1;
00254 
00255                     double px = point3D(0);
00256                     double py = point3D(1);
00257                     double pz = point3D(2);
00258 
00259                     //Transform the 3D point using the transformation matrix Rt
00260                     Eigen::Vector4f  transformedPoint3D = Rt*point3D;
00261 
00262                     //Project the 3D point to the 2D plane
00263                     double inv_transformedPz = 1.0/transformedPoint3D(2);
00264                     double transformed_r,transformed_c; // 2D coordinates of the transformed pixel(r,c) of frame 1
00265                     transformed_c = (transformedPoint3D(0) * fx)*inv_transformedPz + ox; //transformed x (2D)
00266                     transformed_r = (transformedPoint3D(1) * fy)*inv_transformedPz + oy; //transformed y (2D)
00267                     int transformed_r_int = round(transformed_r);
00268                     int transformed_c_int = round(transformed_c);
00269 
00270                     //Asign the intensity value to the warped image and compute the difference between the transformed
00271                     //pixel of frame 1 and the corresponding pixel of frame 2. Compute the error function
00272                     if((transformed_r_int>=0 && transformed_r_int < nRows) &
00273                        (transformed_c_int>=0 && transformed_c_int < nCols))
00274                     {
00275                         //Obtain the pixel values that will be used to compute the pixel residual
00276                         double pixel1; //Intensity value of the pixel(r,c) of the warped frame 1
00277                         double pixel2; //Intensity value of the pixel(r,c) of frame 2
00278                         pixel1 = source_grayImg.at<float>(r,c);
00279                         pixel2 = target_grayImg.at<float>(transformed_r_int,transformed_c_int);
00280 
00281                         //Compute the pixel jacobian
00282                         Eigen::Matrix<double,2,6> jacobianPrRt;
00283                         double temp25 = 1.0/(z+py*temp1+pz*temp2-px*temp3);
00284                         double temp26 = temp25*temp25;
00285 
00286                             //Derivative with respect to x
00287                             jacobianPrRt(0,0)=fx*temp25;
00288                             jacobianPrRt(1,0)=0;
00289 
00290                             //Derivative with respect to y
00291                             jacobianPrRt(0,1)=0;
00292                             jacobianPrRt(1,1)=fy*temp25;
00293 
00294                             //Derivative with respect to z
00295                             jacobianPrRt(0,2)=-fx*(pz*temp4+py*temp5+px*temp11)*temp26;
00296                             jacobianPrRt(1,2)=-fy*(py*temp6+pz*temp9+px*temp14+y)*temp26;
00297 
00298                             //Derivative with respect to yaw
00299                             jacobianPrRt(0,3)=fx*(py*temp7+pz*temp8-px*temp14)*temp25;
00300                             jacobianPrRt(1,3)=fy*(pz*temp4+py*temp5+px*temp15)*temp25;
00301 
00302                             //Derivative with respect to pitch
00303                             jacobianPrRt(0,4)=fx*(py*temp10+pz*temp12-px*temp13)*temp25
00304                             -fx*(-py*temp16-pz*temp17-px*temp24)*(pz*temp4+py*temp5+px*temp11)*temp26;
00305                             jacobianPrRt(1,4)=fy*(py*temp18+pz*temp19-px*temp20)*temp25
00306                             -fy*(-py*temp16-pz*temp17-px*temp24)*(py*temp6+pz*temp9+px*temp14+y)*temp26;
00307 
00308                             //Derivative with respect to roll
00309                             jacobianPrRt(0,5)=fx*(py*temp4+pz*temp21)*temp25
00310                             -fx*(py*temp22-pz*temp23)*(pz*temp4+py*temp5+px*temp11)*temp26;
00311                             jacobianPrRt(1,5)=fy*(pz*temp7+py*temp9)*temp25
00312                             -fy*(py*temp22-pz*temp23)*(py*temp6+pz*temp9+px*temp14+y)*temp26;
00313 
00314                           //Apply the chain rule to compound the image gradients with the projective+RigidTransform jacobians
00315                           Eigen::Matrix<double,1,2> target_imgGradient;
00316                           target_imgGradient(0,0)=target_gradXImg.at<float>(i);
00317                           target_imgGradient(0,1)=target_gradYImg.at<float>(i);
00318                           Eigen::Matrix<double,1,6> jacobian=target_imgGradient*jacobianPrRt;
00319 
00320                           //Assign the pixel residual and jacobian to its corresponding row
00321                           #if ENABLE_OPENMP_MULTITHREADING_ANALYTIC
00322                           #pragma omp critical
00323                           #endif
00324                           {
00325                               jacobians(i,0)=jacobian(0,0);
00326                               jacobians(i,1)=jacobian(0,1);
00327                               jacobians(i,2)=jacobian(0,2);
00328                               jacobians(i,3)=jacobian(0,3);
00329                               jacobians(i,4)=jacobian(0,4);
00330                               jacobians(i,5)=jacobian(0,5);
00331 
00332                               residuals(nCols*transformed_r_int+transformed_c_int,0) = pixel2 - pixel1;
00333                               if(visualizeIterations)
00334                                 warped_source_grayImage.at<float>(transformed_r_int,transformed_c_int) = pixel1;
00335                           }
00336                     }
00337                 }
00338             }
00339         }
00340     }

00342     enum TerminationCriteriaType {NonTerminated = -1,MaxIterationsReached = 0,GradientNormLowerThanThreshold = 1};
00343     bool testTerminationCriteria()
00344     {
00345         bool optimizationFinished = false;
00346 
00347         double gradientNorm = gradients.norm();
00348 
00349         TerminationCriteriaType terminationCriteria = NonTerminated;
00350         if(iter>=max_num_iterations[optimizationLevel])
00351         {
00352             terminationCriteria = MaxIterationsReached;
00353             optimizationFinished = true;
00354         }
00355         else if(gradientNorm<min_gradient_norm[optimizationLevel])
00356         {
00357             terminationCriteria = GradientNormLowerThanThreshold;
00358             optimizationFinished = true;
00359         }
00360 
00361         if(optimizationFinished)
00362         {
00363             #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00364             std::cout<<"----------------------------------------"<<std::endl;
00365             std::cout<<"Optimization level: "<<optimizationLevel<<std::endl;
00366             std::cout<<"Termination criteria: ";
00367             #endif
00368 
00369             switch(terminationCriteria)
00370             {
00371                 case MaxIterationsReached:
00372                     #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00373                     std::cout<<" Max number of iterations reached ("<<max_num_iterations[optimizationLevel]<<")"<<std::endl;;
00374                     #endif
00375                     break;
00376                 case GradientNormLowerThanThreshold:
00377                     #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00378                     std::cout<<" Gradient norm is lower than threshold ("<<gradient_tolerance[optimizationLevel]<<")"<<std::endl;
00379                     #endif
00380                     break;
00381                 default :
00382                     break;
00383             }
00384 
00385             #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00386             std::cout<<"Number iterations: "<<iter<<std::endl;
00387             std::cout<<"gradient norm: "<<gradientNorm<<std::endl;
00388             std::cout<<"----------------------------------------"<<std::endl;
00389             #endif
00390         }
00391 
00392         return optimizationFinished;
00393     }

00457     void optimize()
00458     {
00459         for(optimizationLevel = numOptimizationLevels-1;optimizationLevel>=0;optimizationLevel--)
00460         {
00461             int nRows = gray0Pyr[optimizationLevel].rows;
00462             int nCols = gray0Pyr[optimizationLevel].cols;
00463             int nPoints = nRows * nCols;
00464 
00465             iter = 0;
00466             while(true)
00467             {
00468                 #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00469                 cv::TickMeter tm;tm.start();
00470                 #endif
00471                 cv::Mat warped_source_grayImage;
00472                 if(visualizeIterations)
00473                     warped_source_grayImage = cv::Mat::zeros(nRows,nCols,gray0Pyr[optimizationLevel].type());
00474 
00475                 Eigen::Matrix<double,Eigen::Dynamic,1> residuals;
00476                 residuals = Eigen::MatrixXd::Zero(nPoints,1);
00477                 Eigen::Matrix<double,Eigen::Dynamic,6> jacobians;
00478                 jacobians = Eigen::MatrixXd::Zero(nPoints,6);
00479 
00480                 if(max_num_iterations[optimizationLevel]>0) //compute only if the number of maximum iterations are greater than 0
00481                 {
00482                     computeResidualsAndJacobians(
00483                             gray0Pyr[optimizationLevel],
00484                             depth0Pyr[optimizationLevel],
00485                             gray1Pyr[optimizationLevel],
00486                             gray1GradXPyr[optimizationLevel],
00487                             gray1GradYPyr[optimizationLevel],
00488                             residuals,
00489                             jacobians,
00490                             warped_source_grayImage);
00491 
00492                     gradients = jacobians.transpose()*residuals;
00493                     stateVector = stateVector - lambda_optimization_step[optimizationLevel]*((jacobians.transpose()*jacobians).inverse() * gradients);
00494 
00495                     #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00496                     tm.stop(); std::cout << "Iteration time = " << tm.getTimeSec() << " sec." << std::endl;
00497                     #endif
00498                 }
00499 
00500                 iter++;
00501 
00502                 if(testTerminationCriteria()){break;}
00503 
00504                 if(visualizeIterations)
00505                 {
00506                     cv::Mat imgDiff = cv::Mat::zeros(nRows,nCols,gray1Pyr[optimizationLevel].type());
00507                     cv::absdiff(gray1Pyr[optimizationLevel],warped_source_grayImage,imgDiff);
00508                     cv::imshow("optimize::imgDiff",imgDiff);
00509                     cv::waitKey(0);
00510                 }
00511             }
00512         }
00513 
00514         //After all the optimization process the optimization level is 0
00515         optimizationLevel = 0;
00516 
00517     }
