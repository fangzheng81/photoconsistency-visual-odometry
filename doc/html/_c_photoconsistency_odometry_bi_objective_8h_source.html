00095     void buildPyramid(cv::Mat & img,std::vector<cv::Mat>& pyramid,int levels,bool applyBlur)
00096     {
00097         //Create space for all the images
00098         pyramid.resize(levels);
00099 
00100         double factor = 1;
00101         for(int level=0;level<levels;level++)
00102         {
00103             //Create an auxiliar image of factor times the size of the original image
00104             cv::Mat imgAux;
00105             if(level!=0)
00106             {
00107                 cv::resize(img,imgAux,cv::Size(0,0),factor,factor);
00108             }
00109             else
00110             {
00111                 imgAux = img;
00112             }
00113 
00114             //Blur the resized image with different filter size depending on the current pyramid level
00115             if(applyBlur)
00116             {
00117                 #if ENABLE_GAUSSIAN_BLUR
00118                 if(blurFilterSize[level]>0)
00119                 {
00120                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00121                     cv::GaussianBlur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]),3);
00122                 }
00123                 #elif ENABLE_BOX_FILTER_BLUR
00124                 if(blurFilterSize[level]>0)
00125                 {
00126                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00127                     cv::blur(imgAux,imgAux,cv::Size(blurFilterSize[level],blurFilterSize[level]));
00128                 }
00129                 #endif
00130             }
00131 
00132             //Assign the resized image to the current level of the pyramid
00133             pyramid[level]=imgAux;
00134 
00135             factor = factor/2;
00136         }
00137     }

00140     {
00141         //Compute image gradients
00142         int delta = 0;

00145         //Create space for all the derivatives images
00146         derXPyramid.resize(imagePyramid.size());
00147         derYPyramid.resize(imagePyramid.size());

00201     void computeResidualsAndJacobians(cv::Mat & source_grayImg,
00202                                       cv::Mat & source_depthImg,
00203                                       cv::Mat & target_grayImg,
00204                                       cv::Mat & target_depthImg,
00205                                       cv::Mat & target_intensityGradXImg,
00206                                       cv::Mat & target_intensityGradYImg,
00207                                       cv::Mat & target_depthGradXImg,
00208                                       cv::Mat & target_depthGradYImg,
00209                                       Eigen::Matrix<double,Eigen::Dynamic,1> & residuals,
00210                                       Eigen::Matrix<double,Eigen::Dynamic,6> & jacobians,
00211                                       cv::Mat & warped_source_grayImage)
00212     {
00213         int nRows = source_grayImg.rows;
00214         int nCols = source_grayImg.cols;
00215 
00216         double scaleFactor = 1.0/pow(2,optimizationLevel);
00217         double fx = cameraMatrix(0,0)*scaleFactor;
00218         double fy = cameraMatrix(1,1)*scaleFactor;
00219         double ox = cameraMatrix(0,2)*scaleFactor;
00220         double oy = cameraMatrix(1,2)*scaleFactor;
00221         float inv_fx = 1.f/fx;
00222         float inv_fy = 1.f/fy;
00223 
00224         double x = stateVector[0];
00225         double y = stateVector[1];
00226         double z = stateVector[2];
00227         double yaw = stateVector[3];
00228         double pitch = stateVector[4];
00229         double roll = stateVector[5];
00230 
00231         //Compute the rigid transformation matrix from the parameters
00232         Eigen::Matrix4f Rt = Eigen::Matrix4f::Identity();
00233         double sin_yaw = sin(yaw);
00234         double cos_yaw = cos(yaw);
00235         double sin_pitch = sin(pitch);
00236         double cos_pitch = cos(pitch);
00237         double sin_roll = sin(roll);
00238         double cos_roll = cos(roll);
00239         Rt(0,0) = cos_yaw * cos_pitch;
00240         Rt(0,1) = cos_yaw * sin_pitch * sin_roll - sin_yaw * cos_roll;
00241         Rt(0,2) = cos_yaw * sin_pitch * cos_roll + sin_yaw * sin_roll;
00242         Rt(0,3) = x;
00243         Rt(1,0) = sin_yaw * cos_pitch;
00244         Rt(1,1) = sin_yaw * sin_pitch * sin_roll + cos_yaw * cos_roll;
00245         Rt(1,2) = sin_yaw * sin_pitch * cos_roll - cos_yaw * sin_roll;
00246         Rt(1,3) = y;
00247         Rt(2,0) = -sin_pitch;
00248         Rt(2,1) = cos_pitch * sin_roll;
00249         Rt(2,2) = cos_pitch * cos_roll;
00250         Rt(2,3) = z;
00251         Rt(3,0) = 0;
00252         Rt(3,1) = 0;
00253         Rt(3,2) = 0;
00254         Rt(3,3) = 1;
00255 
00256         depthComponentGain = cv::mean(target_grayImg).val[0]/cv::mean(target_depthImg).val[0];
00257 
00258         #if ENABLE_OPENMP_MULTITHREADING
00259         #pragma omp parallel for
00260         #endif
00261         for (int r=0;r<nRows;r++)
00262         {
00263             for (int c=0;c<nCols;c++)
00264             {
00265                 int i = nCols*r+c; //vector index
00266 
00267                 //Compute the 3D coordinates of the pij of the source frame
00268                 Eigen::Vector4f point3D;
00269                 point3D(2)=source_depthImg.at<float>(r,c);
00270                 if(minDepth < point3D(2) && point3D(2) < maxDepth)//Compute the jacobian only for the valid points
00271                 {
00272                     point3D(0)=(c - ox) * point3D(2) * inv_fx;
00273                     point3D(1)=(r - oy) * point3D(2) * inv_fy;
00274                     point3D(3)=1;
00275 
00276                     double px = point3D(0);
00277                     double py = point3D(1);
00278                     double pz = point3D(2);
00279 
00280                     //Transform the 3D point using the transformation matrix Rt
00281                     Eigen::Vector4f  transformedPoint3D = Rt*point3D;
00282 
00283                     //Project the 3D point to the 2D plane
00284                     double inv_transformedPz = 1.0/transformedPoint3D(2);
00285                     double transformed_r,transformed_c; // 2D coordinates of the transformed pixel(r,c) of frame 1
00286                     transformed_c = (transformedPoint3D(0) * fx)*inv_transformedPz + ox; //transformed x (2D)
00287                     transformed_r = (transformedPoint3D(1) * fy)*inv_transformedPz + oy; //transformed y (2D)
00288                     int transformed_r_int = round(transformed_r);
00289                     int transformed_c_int = round(transformed_c);
00290 
00291                     //Asign the intensity value to the warped image and compute the difference between the transformed
00292                     //pixel of frame 1 and the corresponding pixel of frame 2. Compute the error function
00293                     if((transformed_r_int>=0 && transformed_r_int < nRows) &
00294                        (transformed_c_int>=0 && transformed_c_int < nCols))
00295                     {
00296                         //Obtain the pixel values that will be used to compute the intensity residual
00297                         double intensity1; //Intensity value of the pixel(r,c) of the warped frame 1
00298                         double intensity2; //Intensity value of the pixel(r,c) of frame 2
00299                         intensity1 = source_grayImg.at<float>(r,c);
00300                         intensity2 = target_grayImg.at<float>(transformed_r_int,transformed_c_int);
00301 
00302                         //Obtain the depth values that will be used to the compute the depth residual
00303                         double depth1; //Depth value of the pixel(r,c) of the warped frame 1
00304                         double depth2; //Depth value of the pixel(r,c) of frame 2
00305                         depth1 = source_depthImg.at<float>(r,c);
00306                         depth2 = target_depthImg.at<float>(transformed_r_int,transformed_c_int);
00307 
00308                         //Compute the rigid transformation jacobian
00309                         Eigen::Matrix<double,3,6> jacobianRt;
00310 
00311                             //Derivative with respect to x
00312                             jacobianRt(0,0)=1;
00313                             jacobianRt(1,0)=0;
00314                             jacobianRt(2,0)=0;
00315 
00316                             //Derivative with respect to y
00317                             jacobianRt(0,1)=0;
00318                             jacobianRt(1,1)=1;
00319                             jacobianRt(2,1)=0;
00320 
00321                             //Derivative with respect to z
00322                             jacobianRt(0,2)=0;
00323                             jacobianRt(1,2)=0;
00324                             jacobianRt(2,2)=1;
00325 
00326                             //Derivative with respect to yaw
00327                             jacobianRt(0,3)=py*(-sin(pitch)*sin(roll)*sin(yaw)-cos(roll)*cos(yaw))+pz*(sin(roll)*cos(yaw)-sin(pitch)*cos(roll)*sin(yaw))-cos(pitch)*px*sin(yaw);
00328                             jacobianRt(1,3)=pz*(sin(roll)*sin(yaw)+sin(pitch)*cos(roll)*cos(yaw))+py*(sin(pitch)*sin(roll)*cos(yaw)-cos(roll)*sin(yaw))+cos(pitch)*px*cos(yaw);
00329                             jacobianRt(2,3)=0;
00330 
00331                             //Derivative with respect to pitch
00332                             jacobianRt(0,4)=cos(pitch)*py*sin(roll)*cos(yaw)+cos(pitch)*pz*cos(roll)*cos(yaw)-sin(pitch)*px*cos(yaw);
00333                             jacobianRt(1,4)=cos(pitch)*py*sin(roll)*sin(yaw)+cos(pitch)*pz*cos(roll)*sin(yaw)-sin(pitch)*px*sin(yaw);
00334                             jacobianRt(2,4)=-sin(pitch)*py*sin(roll)-sin(pitch)*pz*cos(roll)-cos(pitch)*px;
00335 
00336                             //Derivative with respect to roll
00337                             jacobianRt(0,5)=py*(sin(roll)*sin(yaw)+sin(pitch)*cos(roll)*cos(yaw))+pz*(cos(roll)*sin(yaw)-sin(pitch)*sin(roll)*cos(yaw));
00338                             jacobianRt(1,5)=pz*(-sin(pitch)*sin(roll)*sin(yaw)-cos(roll)*cos(yaw))+py*(sin(pitch)*cos(roll)*sin(yaw)-sin(roll)*cos(yaw));
00339                             jacobianRt(2,5)=cos(pitch)*py*cos(roll)-cos(pitch)*pz*sin(roll);
00340 
00341                           //Compute the proyective transformation jacobian
00342                           Eigen::Matrix<double,2,3> jacobianProy;
00343 
00344                             //Derivative with respect to x
00345                             jacobianProy(0,0)=fx*inv_transformedPz;
00346                             jacobianProy(1,0)=0;
00347 
00348                             //Derivative with respect to y
00349                             jacobianProy(0,1)=0;
00350                             jacobianProy(1,1)=fy*inv_transformedPz;
00351 
00352                             //Derivative with respect to z
00353                             jacobianProy(0,2)=-(fx*transformedPoint3D(0))*inv_transformedPz*inv_transformedPz;
00354                             jacobianProy(1,2)=-(fy*transformedPoint3D(1))*inv_transformedPz*inv_transformedPz;
00355 
00356                           //Intensity jacobian:
00357                           //Apply the chain rule to compound the intensity gradients with the projective+RigidTransform jacobians
00358                           Eigen::Matrix<double,1,2> target_intensityGradient;
00359                           target_intensityGradient(0,0)=target_intensityGradXImg.at<float>(i);
00360                           target_intensityGradient(0,1)=target_intensityGradYImg.at<float>(i);
00361                           Eigen::Matrix<double,1,6> jacobianItensity=target_intensityGradient*jacobianProy*jacobianRt;
00362 
00363                           //Depth jacobian:
00364                           //Apply the chain rule to compound the depth gradients with the projective+RigidTransform jacobians
00365                           Eigen::Matrix<double,1,2> target_depthGradient;
00366                           target_depthGradient(0,0)=target_depthGradXImg.at<float>(i);
00367                           target_depthGradient(0,1)=target_depthGradYImg.at<float>(i);
00368                           Eigen::Matrix<double,1,6> jacobianRt_z;
00369                           jacobianRt_z(0,0)=jacobianRt(2,0);
00370                           jacobianRt_z(0,1)=jacobianRt(2,1);
00371                           jacobianRt_z(0,2)=jacobianRt(2,2);
00372                           jacobianRt_z(0,3)=jacobianRt(2,3);
00373                           jacobianRt_z(0,4)=jacobianRt(2,4);
00374                           jacobianRt_z(0,5)=jacobianRt(2,5);
00375                           Eigen::Matrix<double,1,6> jacobianDepth=depthComponentGain*(target_depthGradient*jacobianProy*jacobianRt-jacobianRt_z);
00376 
00377                           //Assign the pixel residual and jacobian to its corresponding row
00378                           #pragma omp critical
00379                           {
00380                               //Assign intensity jacobians
00381                               jacobians(i,0)=jacobianItensity(0,0);
00382                               jacobians(i,1)=jacobianItensity(0,1);
00383                               jacobians(i,2)=jacobianItensity(0,2);
00384                               jacobians(i,3)=jacobianItensity(0,3);
00385                               jacobians(i,4)=jacobianItensity(0,4);
00386                               jacobians(i,5)=jacobianItensity(0,5);
00387 
00388                               //Assign intensity residuals
00389                               residuals(nCols*transformed_r_int+transformed_c_int,0) = intensity2 - intensity1;
00390 
00391                               //Assign depth jacobians
00392                               jacobians(2*i,0)=jacobianDepth(0,0);
00393                               jacobians(2*i,1)=jacobianDepth(0,1);
00394                               jacobians(2*i,2)=jacobianDepth(0,2);
00395                               jacobians(2*i,3)=jacobianDepth(0,3);
00396                               jacobians(2*i,4)=jacobianDepth(0,4);
00397                               jacobians(2*i,5)=jacobianDepth(0,5);
00398 
00399                               //Assign depth residuals
00400                               residuals(nCols*2*transformed_r_int+2*transformed_c_int,0) = depthComponentGain*(depth2 - depth1);
00401 
00402                               if(visualizeIterations)
00403                                 warped_source_grayImage.at<float>(transformed_r_int,transformed_c_int) = intensity1;
00404                           }
00405                     }
00406                 }
00407             }
00408         }
00409     }

00411     enum TerminationCriteriaType {NonTerminated = -1,MaxIterationsReached = 0,GradientNormLowerThanThreshold = 1};
00412     bool testTerminationCriteria()
00413     {
00414         bool optimizationFinished = false;
00415 
00416         double gradientNorm = gradients.norm();
00417 
00418         TerminationCriteriaType terminationCriteria = NonTerminated;
00419         if(iter>=max_num_iterations[optimizationLevel])
00420         {
00421             terminationCriteria = MaxIterationsReached;
00422             optimizationFinished = true;
00423         }
00424         else if(gradientNorm<min_gradient_norm[optimizationLevel])
00425         {
00426             terminationCriteria = GradientNormLowerThanThreshold;
00427             optimizationFinished = true;
00428         }
00429 
00430         if(optimizationFinished)
00431         {
00432             #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00433             std::cout<<"----------------------------------------"<<std::endl;
00434             std::cout<<"Optimization level: "<<optimizationLevel<<std::endl;
00435             std::cout<<"Termination criteria: ";
00436             #endif
00437 
00438             switch(terminationCriteria)
00439             {
00440                 case MaxIterationsReached:
00441                     #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00442                     std::cout<<" Max number of iterations reached ("<<max_num_iterations[optimizationLevel]<<")"<<std::endl;;
00443                     #endif
00444                     break;
00445                 case GradientNormLowerThanThreshold:
00446                     #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00447                     std::cout<<" Gradient norm is lower than threshold ("<<gradient_tolerance[optimizationLevel]<<")"<<std::endl;
00448                     #endif
00449                     break;
00450                 default :
00451                     break;
00452             }
00453 
00454             #if ENABLE_PRINT_CONSOLE_OPTIMIZATION_PROGRESS
00455             std::cout<<"Number iterations: "<<iter<<std::endl;
00456             std::cout<<"gradient norm: "<<gradientNorm<<std::endl;
00457             std::cout<<"----------------------------------------"<<std::endl;
00458             #endif
00459         }
00460 
00461         return optimizationFinished;
00462     }

00531         {

00549                 jacobians = Eigen::MatrixXd::Zero(2*nPoints,6);

00551                 if(max_num_iterations[optimizationLevel]>0) //compute only if the number of maximum iterations are greater than 0

00562                             residuals,

00565 
00566                     gradients = jacobians.transpose()*residuals;
00567                     stateVector = stateVector - lambda_optimization_step[optimizationLevel]*((jacobians.transpose()*jacobians).inverse() * gradients);
