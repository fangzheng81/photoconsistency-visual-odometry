00202   void getMatrixNumberRepresentationOf_uint64_t(uint64_t number,cv::Mat & matrixNumberRepresentation)
00203   {
00204 
00205       //Determine the number of digits of the number
00206       int num_digits = 0;
00207       uint64_t number_aux = number;
00208       while(number_aux > 0)
00209       {
00210         num_digits++;
00211         number_aux/=10;
00212       }
00213 
00214       //Compute the matrix representation of the number
00215       matrixNumberRepresentation = cv::Mat::zeros(1,num_digits,CV_8U);
00216 
00217       uint64_t remainder = number;
00218       for(int digitIndex=0;digitIndex<num_digits;digitIndex++)
00219       {
00220         if(remainder==0){break;}
00221 
00222         uint8_t greaterDigit;
00223         uint64_t dividend = remainder;
00224         uint64_t divisor = pow(10,num_digits-1-digitIndex);
00225         uint64_t quotient = remainder / divisor;
00226         greaterDigit = quotient;
00227         matrixNumberRepresentation.at<uint8_t>(0,digitIndex)=greaterDigit;
00228         remainder = dividend - divisor * quotient;
00229       }
00230   }
